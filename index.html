  • HTML ფორმების ელემენტების დეტალური განხილვა
  • სოციალური მედიის ოპტიმიზაციის ტექნიკების შესწავლა

1. შესავალი და თეორიული ნაწილი (30 წუთი)

  • ფორმების რესპონსიული დიზაინი (15 წუთი)
  • რესპონსიული ჩექერ ინსტრუმენტების გამოყენება (10 წუთი)

📚 სრული თეორიული მასალა

🌐 რა არის რესპონსიული ვებ დიზაინი?

რესპონსიული ვებ დიზაინი არის მიდგომა, რომელიც უზრუნველყოფს ვებ გვერდების ოპტიმალურ ჩვენებას ყველა ტიპის მოწყობილობაზე - კომპიუტერიდან დაწყებული, ტაბლეტითა და მობილური ტელეფონით დამთავრებული.

📱 ეკრანის რეზოლუციები და Breakpoint-ები

სტანდარტული Breakpoint-ები:

  • Mobile: 320px - 768px
  • Tablet: 768px - 1024px
  • Desktop: 1024px - 1200px
  • Large Desktop: 1200px+

🎛️ CSS @media Queries

@media სინტაქსი:

@media (პირობა) { CSS წესები }

ძირითადი Media Features:

  • width: ეკრანის სიგანე
  • height: ეკრანის სიმაღლე
  • orientation: ორიენტაცია (portrait/landscape)
  • resolution: ეკრანის რეზოლუცია

📝 HTML ფორმის ელემენტები

ძირითადი Input ტიპები:

  • text: ტექსტური ველი
  • email: ელ-ფოსტის ველი
  • password: პაროლის ველი
  • number: რიცხვითი ველი
  • tel: ტელეფონის ნომერი
  • url: URL მისამართი
  • date: თარიღის არჩევა
  • file: ფაილის ატვირთვა

// ეკრანის ზომის შემოწმება
function checkScreenSize() {
    const width = window.innerWidth;

    if (width <= 480) {
        console.log('მობილური მოწყობილობა');
    } else if (width <= 768) {
        console.log('ტაბლეტი');
    } else if (width <= 1024) {
        console.log('მცირე დესკტოპი');
    } else {
        console.log('დიდი ეკრანი');
    }
}

// ეკრანის ზომის ცვლილების მოსმენა
window.addEventListener('resize', checkScreenSize);

// ფორმის ვალიდაცია
function validateForm() {
    const email = document.getElementById('email');
    const phone = document.getElementById('phone');

    // ელ-ფოსტის ვალიდაცია
    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailPattern.test(email.value)) {
        alert('გთხოვთ შეიყვანოთ სწორი ელ-ფოსტა');
        return false;
    }

    // ტელეფონის ვალიდაცია (ქართული ფორმატი)
    const phonePattern = /^(\+995|995)?[0-9]{9}$/;
    if (!phonePattern.test(phone.value.replace(/\s/g, ''))) {
        alert('გთხოვთ შეიყვანოთ სწორი ტელეფონის ნომერი');
        return false;
    }

    return true;
}

// Touch events მობილური მოწყობილობებისთვის

// Media Query-ების JavaScript-ით შემოწმება
const mediaQuery = window.matchMedia('(max-width: 768px)');

function handleTabletChange(e) {
    if (e.matches) {
        // ტაბლეტის ეკრანზე გამოსაჩენი ფუნქციონალი
        console.log('ტაბლეტის რეჟიმი აქტიურია');
    } else {
        // დესკტოპის ეკრანზე გამოსაჩენი ფუნქციონალი
        console.log('დესკტოპის რეჟიმი აქტიურია');
    }
}

mediaQuery.addListener(handleTabletChange);
handleTabletChange(mediaQuery);
